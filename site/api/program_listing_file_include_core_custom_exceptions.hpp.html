#pragma once
#include <exception>
#include <stdexcept>
#include <string>
#include <sstream>
#include <iostream>


class UnimplementedError : public std::runtime_error
{
    private:
    std::string putTogether(const char* file, const char* function, const int line,
    const std::string what) noexcept
    {
        std::stringstream message;
        message << file << ":" << function << ":" <<
         std::to_string(line) << ": UNIMPLEMENTED ERROR:" << what << std::endl;
        return message.str();
    }



    public:
    UnimplementedError(const std::string& what) : std::runtime_error(what) {}
    UnimplementedError(const char* file, const char* function, const int line,
                        const std::string what  ) : runtime_error(putTogether(file,function,line,what)){}

};
