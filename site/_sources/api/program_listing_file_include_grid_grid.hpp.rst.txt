   template <QuantityType Q>
       Q Grid2d<Q>::getQuantity(size_t i, size_t j) const
       {
           size_t inner_index = i * (n_horizontal+2*n_horizontal_ghost) + j;
           size_t blocksize = (n_horizontal+2*n_horizontal_ghost)*(n_radial+2*n_radial_ghost);
   
           std::array<D,N> data;
           for (size_t n=0; n< N; n++)
           {
               data[n] = grid[n*blocksize + inner_index];
           }
           return Q{data};
       }
   template <QuantityType Q>
       void Grid2d<Q>::setQuantity(Q value, size_t i, size_t j)
       {
           size_t inner_index = i * (n_horizontal+2*n_horizontal_ghost) + j;
           size_t blocksize = (n_horizontal+2*n_horizontal_ghost)*(n_radial+2*n_radial_ghost);
   
           for (size_t n=0; n< N; n++)
           {
               grid[n*blocksize + inner_index] = value[n];
           }
       }
   
   template <typename D>
   class GridDimensions2d
   {
       private:
       size_t n_radial_total;        
       size_t n_horizontal_total;    
       std::vector<D> radial_coordinates; 
       std::vector<D> horizontal_coordinates; 
   
       public:
       GridDimensions2d() : n_radial_total(0), n_horizontal_total(0) {};
       GridDimensions2d(size_t n_r, size_t n_h) : n_radial_total(n_r), n_horizontal_total(n_h), radial_coordinates(n_r), horizontal_coordinates(n_h) {};
   
       size_t getNradial() const {return this->n_radial_total;}
       size_t getNhorizontal() const {return this->n_horizontal_total;} 
   
       const std::vector<D>& getRadialCoordinates() const { return radial_coordinates; }
       const std::vector<D>& getHorizontalCoordinates() const { return horizontal_coordinates; }
   
       D& radialCoord(size_t i) { return radial_coordinates[i]; }
       D& radialCoord(size_t i) const { return radial_coordinates[i]; }
       D& horizontalCoord(size_t j) { return horizontal_coordinates[j]; }
       D& horizontalCoord(size_t j) const { return horizontal_coordinates[j]; }
   
       inline D interfaceAreaIphalf(size_t i,size_t j);
       inline D interfaceAreaJphalf(size_t i,size_t j);
       inline D cellVolume(size_t i, size_t j);
   
       static GridDimensions2d createUniformGrid(size_t n_r, size_t n_h, D r0, D rN, D theta0, D thetaN)
       {
           GridDimensions2d grid_dims(n_r, n_h);
           for(size_t i=0; i< grid_dims.getNradial(); i++)
           {
               grid_dims.radialCoord(i) = r0 + static_cast<D>(i)*(rN-r0)/(grid_dims.getNradial()-1);
           }
           for(size_t j=0; j< grid_dims.getNhorizontal(); j++)
           {
               grid_dims.horizontalCoord(j) = theta0 + static_cast<D>(j)*(thetaN-theta0)/static_cast<D>(grid_dims.getNhorizontal()-1);
           }
           return grid_dims;
       }
   
   };
   
   template<typename D>
    inline D GridDimensions2d<D>::interfaceAreaIphalf(size_t i,size_t j)
    {
       /*
        * At i+1/2 the area is:
        * 2 pi r_{i+1/2}^2 \sin \theta_j \Delta \theta
        */
       if(i == 0 || j== 0 || i >= n_radial_total-1 || j >= n_horizontal_total-1)
       {
           fprintf(stderr, "Interface area at i=%zu, j=%zu is undefined\n", i, j);
           return 0.0; // No area at the boundaries
       }
       D delta_theta = 0.5*(this->horizontal_coordinates[j+1] - this->horizontal_coordinates[j-1]);
       //D theta_iphalf = 0.5*(this->horizontal_coordinates[j+1] + this->horizontal_coordinates[j]);
       //D theta_imhalf = 0.5*(this->horizontal_coordinates[j-1] + this->horizontal_coordinates[j]);
       D r_iphalf = 0.5*(this->radial_coordinates[i+1] + this->radial_coordinates[i]);
       //D r_imhalf = 0.5*(this->radial_coordinates[i-1] + this->radial_coordinates[i]);
       return 2 * M_PI * r_iphalf * r_iphalf
        * std::sin(this->horizontal_coordinates[j]) * delta_theta;
       //return 2 * constants::kPi * (r_iphalf*r_iphalf) * (std::cos(theta_iphalf) - std::cos(theta_imhalf));
   
         
    }
   
   
   template<typename D>
    inline D GridDimensions2d<D>::interfaceAreaJphalf(size_t i,size_t j)
    {
   
       if(j == 0 || j >= n_horizontal_total-1)
       {
           fprintf(stderr, "Interface area at i=%zu, j=%zu is undefined\n", i, j);
           return 0.0; // No area at the boundaries
       }
       D delta_r = 0.5*(this->radial_coordinates[i+1] - this->radial_coordinates[i-1]);
       D r_iphalf = 0.5*(this->radial_coordinates[i+1] + this->radial_coordinates[i]);
       D r_imhalf = 0.5*(this->radial_coordinates[i-1] + this->radial_coordinates[i]);
       D theta_jphalf =  0.5*(this->horizontal_coordinates[j+1] + this->horizontal_coordinates[j]);
       return 2 * M_PI * this->radial_coordinates[i] * std::sin(theta_jphalf) * delta_r;
       return  constants::kPi * std::sin(theta_jphalf) * (r_iphalf + r_imhalf)*(r_iphalf-r_imhalf);
      
    }

       {

           auto n_r = grid.getNr();

           for( size_t j =0; j<n_ghost_h; j++)

               {
                   grid.setQuantity(grid.getQuantity(i,n_ghost_h+j), i, n_h+n_ghost_h+j);
                   grid.setQuantity(grid.getQuantity(i,n_h+j), i, j);
               }
           }
           for(size_t i=0; i< n_ghost_r; i++)
           {
               for(size_t j=0; j< n_h + 2*n_ghost_h; j++)
               {
                   //grid.setQuantity(grid.getQuantity(n_ghost_r+i,j), n_r+n_ghost_r+i, j);
                   //grid.setQuantity(grid.getQuantity(n_r+i,j), i, j);
                   grid.setQuantity(grid.getQuantity(n_ghost_r,j), i, j);
                   grid.setQuantity(grid.getQuantity(n_r+n_ghost_r-1,j), n_r+n_ghost_r+i, j);
               }
           }
   
       }
