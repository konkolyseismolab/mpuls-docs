template<QuantityType Q>
class TimeSolverBase
{
    using D = typename QuantityTraits<Q>::D;
    public:
    virtual ~TimeSolverBase() {};

    virtual void solve() = 0;

    virtual void setT(D new_T) = 0;

    virtual D getT() const = 0;

    virtual void setT0(D new_t0) = 0;

    virtual D getT0() const = 0;

    virtual void setDt(D new_dt) = 0;

    virtual D getDt() const = 0;
};

template<class T, class Q>
concept TimeSolverType = std::derived_from<T, TimeSolverBase<Q>> &&
                         (! std::same_as<T, TimeSolverBase<Q>>);

template<class T>
concept HasTimeSolverInterface = requires(T& t) {
    { t.solve() };
    { t.setT(0.0) };
    { t.getT() } -> std::convertible_to<double>;
    { t.setT0(0.0) };
    { t.getT0() } -> std::convertible_to<double>;
    { t.setDt(0.0) };
    { t.getDt() } -> std::convertible_to<double>;
};


template <QuantityType Q, class RHS, class CFL, class Boundary>
class SSP_RK2 : public TimeSolverBase<Q>
{
    private:
    using D = typename QuantityTraits<Q>::D;

    RHS& rhs;
    CFL& cfl;
    Boundary& boundary;
    D dt;
    D T;
    D t0;
    Grid2d<Q>& old_grid;
    Grid2d<Q>& new_grid;
    Grid2d<Q> temp_grid;
    Grid2d<Q> k1_grid;



    public:
    SSP_RK2(D dt, D T, D t0,Grid2d<Q>& o_grid,Grid2d<Q>&n_grid, RHS& rhs, CFL& cfl, Boundary& boundary) :
      TimeSolverBase<Q>(), rhs(rhs), cfl(cfl), boundary(boundary), dt(dt), T(T), t0(t0), old_grid(o_grid), new_grid(n_grid)
    {
        temp_grid = Grid2d<Q>(o_grid.getNr(), o_grid.getNh(), o_grid.getNrGhost(), o_grid.getNhGhost());
        k1_grid = Grid2d<Q>(o_grid.getNr(), o_grid.getNh(), o_grid.getNrGhost(), o_grid.getNhGhost());
    }
    void setDt(D new_dt) override { dt = new_dt; }
    void setT(D new_T) override { T = new_T; }
    void setT0(D new_t0) override { t0 = new_t0; }
    D getT0() const override { return t0; }
    D getDt() const override { return dt; }
    D getT() const override {return T;}

    void step_1(size_t i, size_t j)
    {
        Q k1;
        Q temp;
        Q old = old_grid.getQuantity(i,j);
        k1 = rhs(old_grid,i,j) * dt;
        temp = old + 1.0*k1;
        k1_grid.setQuantity(k1,i,j);
        temp_grid.setQuantity(temp,i,j);


    }

    void step_2(size_t i, size_t j)
    {
        Q old = old_grid.getQuantity(i,j);
        Q k1 =k1_grid.getQuantity(i,j);
        //Q temp = temp_grid.getQuantity(i,j);
        Q k2;
        k2 = rhs(temp_grid,i,j) * dt ;
        new_grid.setQuantity(old + 1. * (1.*k1+0.*k2),i,j);
    }

    void solve() override
    {
        D t_current = t0;
        size_t i_start = old_grid.getNrGhost();
        size_t j_start = old_grid.getNhGhost();
        size_t i_end = old_grid.getNr()+ old_grid.getNrGhost();
        size_t j_end = old_grid.getNh()+ old_grid.getNhGhost();
        dt = cfl.getNewDt(old_grid);
        auto start_time = std::chrono::high_resolution_clock::now();
        int i=0;
        while(t_current < T)
        {
            if(t_current + dt > T ) dt = T - t_current;
            for(auto i = i_start; i<i_end; i++ )
                for(auto j= j_start; j<j_end; j++)
                {
                    step_1(i,j);
                }
            boundary.setConditions(temp_grid);
            for(auto i = i_start; i< i_end; i++)
                for(auto j = j_start; j< j_end; j++)
                {
                    step_2(i,j);
                }
            t_current += dt;
            boundary.setConditions(new_grid);
            dt = cfl.getNewDt(new_grid);
            std::swap(old_grid,new_grid);
            fprintf(stderr, "\rCurrent time: %e, dt: %e, step: %d", t_current, dt,++i);

        }
        auto end_time = std::chrono::high_resolution_clock::now();
        std::chrono::duration<double, std::milli> elapsed = end_time - start_time;
        std::fprintf(stderr,", total time: %lf ms and %lf ms/step\n", elapsed.count(), elapsed.count() / static_cast<double>(i));
    }
};

template<QuantityType Q, class FluxOperatorTypeRad,class FluxOperatorTypeHor, class SourceType, EquationType<Q> Equation>
class RightHandSide
{
    FluxOperatorTypeRad& f_r;
    FluxOperatorTypeHor& f_t;
    SourceType& source_term;
    Equation& eq;

    public:
    RightHandSide(FluxOperatorTypeRad& f_r, FluxOperatorTypeHor& f_t, SourceType& source_term, Equation& eq) :
     f_r(f_r), f_t(f_t), source_term(source_term), eq(eq) {}

    Q operator()(const Grid2d<Q>& grid, size_t i, size_t j)
    {
        Q rhs;
        rhs = - f_r(grid,i,j) - f_t(grid,i,j) + source_term(grid, i, j);
        return rhs;
    }
};

template<class RHS, class NewOp,QuantityType Q>
class RightHandSideDecorator
{
    RHS& original;
    NewOp& new_operator;
    public:
    RightHandSideDecorator(RHS& orig, NewOp& new_op) : original(orig), new_operator(new_op) {}

    Q operator()(const Grid2d<Q>& grid, size_t i, size_t j)
    {
        Q original_rhs = original(grid,i,j);
        Q new_operator_rhs = new_operator(grid,i,j);
        return original_rhs + new_operator_rhs;
    }
};


template<QuantityType Q, class Equation>
class SimpleSource
{
    using D = typename QuantityTraits<Q>::D;
    static constexpr int N = QuantityTraits<Q>::N;
    Equation& eq;
    GridDimensions2d<D>& grid_dim;

    public:
    SimpleSource(Equation& eq, GridDimensions2d<D>& grid) : eq(eq), grid_dim(grid) {}
    Q operator()(const Grid2d<Q>& grid,  size_t i, size_t j)
    {
        Q q_ij = grid.getQuantity(i,j);
        D radius = grid_dim.radialCoord(i);
        D theta = grid_dim.horizontalCoord(j);
        Q source = eq.computeSource(q_ij, std::array<D,2>{radius,theta});
        D delta_theta = 0.5 * (grid_dim.horizontalCoord(j+1) + grid_dim.horizontalCoord(j-1));;
        if constexpr (N==4)
        {
            source[2] *= 2*std::sin(0.5*delta_theta)/delta_theta;
        }
        return source;
    }
};


template<QuantityType Q, class Equation>
class CFL
{
    using D = typename QuantityTraits<Q>::D;
    D cfl_number;
    Equation& eq;
    GridDimensions2d<D>& grid_dim;
    public:
    CFL(D cfl, Equation& eq, GridDimensions2d<D>& grid_dim) : cfl_number(cfl), eq(eq), grid_dim(grid_dim) {}

    D getNewDt(const Grid2d<Q>& grid)
    {
        auto n_r = grid.getNr();
        auto n_h = grid.getNh();
        auto n_r_g = grid.getNrGhost();
        auto n_h_g = grid.getNhGhost();
        D max_cond_radial = 0.0;
        D max_cond_horizontal = 0.0;

        for(size_t i = n_r_g; i< n_r + n_r_g; i++)
        {
            for(size_t j = n_h_g; j< n_h + n_h_g; j++)
            {
                Q q_ij = grid.getQuantity(i,j);
                D radius = grid_dim.radialCoord(i);
                D theta = grid_dim.horizontalCoord(j);
                D speed_radial = eq.getMaxVelocity(q_ij, Direction::Radial,std::array<D,2>{radius,theta});
                D speed_horizontal = eq.getMaxVelocity(q_ij, Direction::Horizontal,std::array<D,2>{radius,theta});
                D delta_r = 0.5*(grid_dim.radialCoord(i+1) - grid_dim.radialCoord(i-1));
                D delta_theta = 0.5*(grid_dim.horizontalCoord(j+1) - grid_dim.horizontalCoord(j-1));
                D hor_size = grid_dim.radialCoord(i) * delta_theta;

                D cond_radial = speed_radial / delta_r;
                D cond_horizontal = speed_horizontal / hor_size;
                max_cond_radial = std::max(max_cond_radial, cond_radial);
                max_cond_horizontal = std::max(max_cond_horizontal, cond_horizontal);
            }
        }
        D dt = cfl_number / std::max(max_cond_radial, max_cond_horizontal);
        return dt;
    }

};
