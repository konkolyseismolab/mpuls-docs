#include "io/profile_writer.hpp"
#include "H5Cpp.h"
#include <sstream>


void writeProfileToHdf5(const std::string& filename,
                           const std::vector<double>& radius,
                           const std::vector<double>& kappa,
                           const std::vector<double>& rho,
                           const std::vector<double> &q)
{
    using namespace H5;

    const hsize_t size = radius.size();

// H5F_ACC_TRUNC means "truncate" mode:
// If the file already exists, it will be deleted and recreated.
// If the file does not exist, it will be created.
// This ensures that we always start with a clean, fresh HDF5 file.
    H5File file(filename, H5F_ACC_TRUNC);
    hsize_t dims[1] = {size};

// Define a 1D dataspace with 'size' elements.
// This describes the shape of the dataset we are about to create.
    DataSpace dataspace(1, dims);


// Create a dataset named "r" with native double-precision type (PredType::NATIVE_DOUBLE) and the given dataspace.
// 'PredType' defines the HDF5 data type; 'NATIVE_DOUBLE' matches the system's native C++ double format.
// Then write the contents of the 'r' vector into it using a raw pointer.
    file.createDataSet("radius", PredType::NATIVE_DOUBLE, dataspace).write(radius.data(), PredType::NATIVE_DOUBLE);
    file.createDataSet("kappa", PredType::NATIVE_DOUBLE, dataspace).write(kappa.data(), PredType::NATIVE_DOUBLE);
    file.createDataSet("rho", PredType::NATIVE_DOUBLE, dataspace).write(rho.data(), PredType::NATIVE_DOUBLE);
    file.createDataSet("q", PredType::NATIVE_DOUBLE, dataspace).write(q.data(), PredType::NATIVE_DOUBLE);

}

void write2DScalarProfileToHdf5(const std::string &filename,
                           const std::vector<double> &radius,
                           const std::vector<double> &theta,
                           const std::vector<double> &scalar_field)
{
    using namespace H5;

    const hsize_t radial_size = radius.size();
    const hsize_t theta_size = theta.size();

    H5File file(filename, H5F_ACC_TRUNC);
    hsize_t radial_dims[1] = {radial_size};
    hsize_t theta_dims[1] = {theta_size};
    hsize_t scalar_dims[2] = {radial_size, theta_size};

    DataSpace radial_dataspace(1, radial_dims);
    DataSpace theta_dataspace(1, theta_dims);
    DataSpace scalar_dataspace(2, scalar_dims);

    file.createDataSet("radius", PredType::NATIVE_DOUBLE, radial_dataspace).write(radius.data(), PredType::NATIVE_DOUBLE);
    file.createDataSet("theta", PredType::NATIVE_DOUBLE, theta_dataspace).write(theta.data(), PredType::NATIVE_DOUBLE);
    file.createDataSet("scalar_field", PredType::NATIVE_DOUBLE, scalar_dataspace).write(scalar_field.data(), PredType::NATIVE_DOUBLE);

}
