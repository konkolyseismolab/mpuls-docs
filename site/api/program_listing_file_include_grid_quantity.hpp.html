#pragma once
#include <array>
#include <cstdlib>
template <typename D, int N>
class QuantityBase
{
    private:
        std::array<D,N> state_vector;

    public:
    QuantityBase() : state_vector(std::array<D,N>{0}) {};
    QuantityBase(std::array<D,N> input) : state_vector(input) {} ;

    D& operator[](size_t n)
    {
        return this->state_vector[n];
    }

    const D&  operator[](const size_t n) const
    {
        return this->state_vector[n];
    }
    QuantityBase operator+(const QuantityBase& other) const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = this->state_vector[n] + other.state_vector[n];
        }
        return result;
    }

    QuantityBase operator-(const QuantityBase& other) const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = this->state_vector[n] - other.state_vector[n];
        }
        return result;
    }

    QuantityBase operator*(D scalar) const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = this->state_vector[n] * scalar;
        }
        return result;
    }

    QuantityBase operator/(D scalar) const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = this->state_vector[n] / scalar;
        }
        return result;
    }

    QuantityBase operator/(QuantityBase other) const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = this->state_vector[n] / other.state_vector[n];
        }
        return result;
    }
    QuantityBase operator*(QuantityBase other) const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = this->state_vector[n] * other.state_vector[n];
        }
        return result;
    }
    QuantityBase operator+(D scalar) const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = this->state_vector[n] + scalar;
        }
        return result;
    }
    QuantityBase operator-() const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = - this->state_vector[n];
        }
        return result;
    }

    QuantityBase abs() const
    {
        QuantityBase result;
        for (size_t n=0; n< N; n++)
        {
            result[n] = std::abs(this->state_vector[n]);
        }
        return result;
    }
    void FixZeroDivider()
    {
        for (size_t n=0; n< N; n++)
        {
            if (std::abs(this->state_vector[n]) < 1e-16)
                this->state_vector[n] = 1e-16;
        }
    }
};
template <typename D, int N>
QuantityBase<D,N> operator*(D scalar, const QuantityBase<D,N>& quantity)
{
    return quantity * scalar;
};

template <typename D, int N>
QuantityBase<D,N> operator+(D scalar, const QuantityBase<D,N>& quantity)
{
    return quantity + scalar;
};

template<class Q>
struct QuantityTraits
{
    using D = double;
    static constexpr int N = 0;
};

{
    using D = Doub;

};
