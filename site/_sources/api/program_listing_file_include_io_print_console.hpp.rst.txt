     inline void initializeRun(size_t width = 60) {
       std::cout << "\n\n";
   
       std::string line(width, '=');
       std::cout << line << "\n";
   
       auto printCentered = [width](const std::string& text) {
           size_t padding = (width > text.size()) ? (width - text.size()) / 2 : 0;
           std::cout << std::string(padding, ' ') << text << "\n";
       };
   
       printCentered("MPULS - Modules for Stellar Pulsations - 1D module");
       printCentered("Version: " + mpuls::version);
       printCentered("Author: " + mpuls::author);
   
       std::cout << line << "\n\n";
     }
   
   
   
       inline void placeholder_warning(const std::string& parameter,
                                       const std::string& file,
                                       const std::string& function)
       {
           static std::unordered_set<std::string> already_reported;
   
           // If we've already warned about this parameter, do nothing
           if (already_reported.count(parameter) > 0)
               return;
   
           // Otherwise print the warning and remember it
           already_reported.insert(parameter);
   
           std::cout << "[WARNING] Placeholder parameter used: " << parameter << "\n"
                   << "          Location: " << file << " :: " << function << "\n"
                   << "          Replace this with the final implementation later.\n";
       }
   
   
       inline void unimplemented_warning(const std::string& function_name,
                                     const std::string& file)
       {
           static std::unordered_set<std::string> reported;
   
           std::string key = function_name + "|" + file;
   
           if (reported.count(key) > 0)
               return;
   
           reported.insert(key);
   
           std::cout << "[UNIMPLEMENTED] Function not implemented: " << function_name << "\n"
                     << "                Location: " << file <<"\n"
                     << "                Implement this function later.\n";
       }
