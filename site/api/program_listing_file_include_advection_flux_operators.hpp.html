template<QuantityType Q>
class LimiterBase
{
    using D = typename QuantityTraits<Q>::D;
    public:
    virtual ~LimiterBase() {};
    virtual Q operator()(const Q& delta_plus, const Q& delta_minus) const = 0;
};

template<class T, typename Q>
concept LimiterType = std::derived_from<T,LimiterBase<Q>> && (! std::same_as<T,LimiterBase<Q>>);

template<QuantityType Q>
class VanLeer : public LimiterBase<Q>
{
    using D = typename QuantityTraits<Q>::D;
    static constexpr int N = QuantityTraits<Q>::N;
    inline D single_var(const D delta_plus, const D delta_minus) const
    {
        if(delta_plus*delta_minus > 0)

            return 2 * delta_minus/(delta_plus + delta_minus);
        else
            return 0;
    }
    public:
    VanLeer() : LimiterBase<Q>() {}
    Q operator()(const Q& delta_plus, const Q& delta_minus) const override
    {
        Q retval;
        for(auto n=0; n<N; n++)
        {
            retval[n] = single_var(delta_plus[n],delta_minus[n]);
        }
        return retval;
    }
};

template<QuantityType Q>
class Superbee : public LimiterBase<Q>
{
    using D = typename QuantityTraits<Q>::D;
    static constexpr int N = QuantityTraits<Q>::N;
    inline D single_var(const D delta_plus, const D delta_minus) const
    {
        D multiplier = delta_plus*delta_minus > 0 ? 2.: 0.;
        D min1 =std::min(std::abs(delta_plus),std::abs(delta_minus));
        D min2 = 0.5* std::max(std::abs(delta_plus),std::abs(delta_minus));
        D val =multiplier* std::min(min1,min2) ;
        return  std::abs(delta_plus) > 1e-16 ? val/delta_plus : 0;
    }
    public:
    Superbee() : LimiterBase<Q>() {}

    Q operator()(const Q& delta_plus, const Q& delta_minus) const override
    {
        Q retval;
        for (int n=0; n<N; n++)
        {
            retval[n] = single_var(delta_plus[n],delta_minus[n]);
        }
        return retval;
    }
};


template<QuantityType Q>
class MinMod : public LimiterBase<Q>
{
    using D = typename QuantityTraits<Q>::D;
    static constexpr int N = QuantityTraits<Q>::N;
    inline D single_var(const D delta_plus, const D delta_minus) const
    {
        if( std::abs(delta_plus) > 1e-16)
        return (std::min(std::max(delta_minus,0.),std::max(delta_plus,0.))
        +std::max(std::min(delta_minus,0.), std::min(delta_plus,0.)))/delta_plus;
        else
         return 0.;
    }
public:
    MinMod() : LimiterBase<Q>() {}
    Q operator()(const Q& delta_plus, const Q& delta_minus) const override
    {
        Q retval;
        for (int n=0; n<N; n++)
        {
            retval[n] = single_var(delta_plus[n],delta_minus[n]);
        }
        return retval;
    }
};

template<QuantityType Q>
class ReconstructionBase
{
    public:
    virtual ~ReconstructionBase() {}

    virtual Q get_left_state(const Grid2d<Q>& grid, size_t i, size_t j) = 0;

    virtual Q get_right_state(const Grid2d<Q>& grid, size_t i, size_t j) = 0;
};

template<class T, class Q>
concept ReconstructionType = std::derived_from<T, ReconstructionBase<Q>> && (!std::same_as<T, ReconstructionBase<Q>>);

        }
    }

template<QuantityType Q, Direction Dir, EquationType<Q> Eq>
class LLFflux
{
    private:
    using D = typename QuantityTraits<Q>::D;
    Eq& eq;

    public:
    LLFflux(Eq& equation) : eq(equation) {}

    Q operator()(const Q& q_plus, const Q& q_minus, std::span<const D> coords) const
    {
        Q flux_plus = eq.computeFlux(q_plus,Dir, coords);
        Q flux_minus = eq.computeFlux(q_minus,Dir, coords);
        D max_speed = std::max(eq.getMaxVelocity(q_plus,Dir,coords), eq.getMaxVelocity(q_minus,Dir, coords));
        return 0.5 * (flux_plus + flux_minus) - 0.5 * max_speed * (q_plus - q_minus);
    }
};

template<QuantityType Q, Direction Dir, EquationType<Q> Eq>
class CIRFlux
{
    private:
    using D = typename QuantityTraits<Q>::D;
    Eq& eq;

    public:
    CIRFlux(Eq& equation) : eq(equation) {}


    Q operator()(const Q& q_plus, const Q& q_minus,std::span<const D> coords) const
    {
        Q flux_plus = eq.computeFlux(q_plus,Dir,coords);
        Q flux_minus = eq.computeFlux(q_minus,Dir,coords);
        D speed_plus = eq.getSpeed(q_plus,Dir,coords);
        D speed_minus = eq.getSpeed(q_minus,Dir, coords);
        return speed_plus > 0 && speed_minus > 0 ? flux_minus : (speed_plus < 0 && speed_minus < 0 ? flux_plus : Q());
    }
};

template <QuantityType Q, ReconstructionType<Q> Reconst, class Fluxcalc>
class FluxOperator
{
  private:
    using D = typename QuantityTraits<Q>::D;
    static constexpr int N = QuantityTraits<Q>::N;
    Direction direction;
    GridDimensions2d<D>& grid_dims;
    Reconst& reconst;
    Fluxcalc& fluxcalc;

    public:
    FluxOperator(Direction dir, GridDimensions2d<D>& dims, Reconst& rec, Fluxcalc& calc) : direction(dir), grid_dims(dims), reconst(rec),
    fluxcalc(calc) {}

    Q operator()(const Grid2d<Q>& grid, size_t i, size_t j)
    {
        D A_phalf, A_mhalf;
        Q q_phalf_plus, q_phalf_minus, q_mhalf_plus, q_mhalf_minus;
        D cell_volume = grid_dims.cellVolume(i,j);
        D r_phalf, r_mhalf, theta_phalf, theta_mhalf;
        if(direction == Direction::Radial)
        {
            A_phalf = grid_dims.interfaceAreaIphalf(i,j);
            A_mhalf = grid_dims.interfaceAreaIphalf(i-1,j);
            q_phalf_plus = reconst.get_left_state(grid,i+1,j);
            q_phalf_minus = reconst.get_right_state(grid,i,j);
            q_mhalf_plus = reconst.get_left_state(grid,i,j);
            q_mhalf_minus = reconst.get_right_state(grid,i-1,j);
            r_phalf = 0.5 * (grid_dims.radialCoord(i) +grid_dims.radialCoord(i+1));
            r_mhalf = 0.5 * (grid_dims.radialCoord(i) +grid_dims.radialCoord(i-1));
            theta_phalf = theta_mhalf = grid_dims.horizontalCoord(j);

        }
        else
        {
            A_phalf = grid_dims.interfaceAreaJphalf(i,j);
            A_mhalf = grid_dims.interfaceAreaJphalf(i,j-1);
            q_phalf_plus = reconst.get_left_state(grid,i,j+1);
            q_phalf_minus = reconst.get_right_state(grid,i,j);
            q_mhalf_plus = reconst.get_left_state(grid,i,j);
            q_mhalf_minus = reconst.get_right_state(grid,i,j-1);
            theta_phalf = 0.5 * (grid_dims.horizontalCoord(j) + grid_dims.horizontalCoord(j+1));
            theta_mhalf = 0.5 * (grid_dims.horizontalCoord(j) + grid_dims.horizontalCoord(j-1));
            r_phalf = r_mhalf = grid_dims.radialCoord(i);
        }

        // --- Calculate the fluxes at the reconstructed interfaces ---
        Q flux_phalf = fluxcalc(q_phalf_plus, q_phalf_minus,std::array<D,2>{r_phalf,theta_phalf}) * A_phalf;
        Q flux_mhalf = fluxcalc(q_mhalf_plus, q_mhalf_minus,std::array<D,2>{r_mhalf,theta_mhalf}) * A_mhalf;

        return (flux_phalf - flux_mhalf) / cell_volume;
    }

};
